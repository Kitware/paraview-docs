33 # This script tests the ParaView install tree by building the examples
34 #
35 # The script expects the following input variables:
36 # PARAVIEW_BINARY_DIR : Build path for ParaView (To run 'make install' here)
37 # PARAVIEW_INSTALL_DIR : Install path for ParaView Examples are built against
38 # (This should be CMAKE_INSTALL_PREFIX set for ParaView)
39 # PARAVIEW_SOURCE_DIR : ParaView source dir (For source location of Examples)
40 # PARAVIEW_TEST_DIR : Temporary directory for location of Examples build tree
41 # PARAVIEW_VERSION : ParaView version string used when creating the installtree
42 
43 message (STATUS "Building Examples against ParaView install tree")
44 set (ParaView_DIR
45  $ENV{DESTDIR}${PARAVIEW_INSTALL_DIR}/lib/cmake/paraview-${PARAVIEW_VERSION})
46 
47 # Build target "INSTALL" for paraview
48 if (MSVC)
49  set (install_tgt "INSTALL")
50  set (opt "/maxcpucount")
51 else ()
52  set (install_tgt "install")
53  set (opt "-j5")
54 endif ()
55 execute_process (COMMAND ${CMAKE_COMMAND}
56  --build ${PARAVIEW_BINARY_DIR}
57  --target ${install_tgt}
58  -- ${opt}
59  WORKING_DIRECTORY ${PARAVIEW_BINARY_DIR}
60  RESULT_VARIABLE irv)
61 if (NOT irv EQUAL 0)
62  message(FATAL_ERROR "Could not build target 'install' for ParaView")
63 endif ()
64 
65 set (INSTALL_TEST_BUILD_DIR ${PARAVIEW_TEST_DIR}/Examples-bld)
66 if (NOT EXISTS ${INSTALL_TEST_BUILD_DIR})
67  execute_process(COMMAND ${CMAKE_COMMAND} -E
68  make_directory ${INSTALL_TEST_BUILD_DIR})
69 endif ()
70 execute_process (
71  COMMAND ${CMAKE_COMMAND}
72  -DParaView_DIR:PATH=${ParaView_DIR}
73  ${PARAVIEW_SOURCE_DIR}/Examples
74  WORKING_DIRECTORY ${INSTALL_TEST_BUILD_DIR}
75  RESULT_VARIABLE crv)
76 if (NOT crv EQUAL 0)
77  message(FATAL_ERROR "Configuration failed with return code ${crv}")
78 endif ()
79 execute_process (
80  COMMAND ${CMAKE_COMMAND} --build ${INSTALL_TEST_BUILD_DIR} --clean-first
81  WORKING_DIRECTORY ${INSTALL_TEST_BUILD_DIR}
82  RESULT_VARIABLE rv)
83 if (NOT rv EQUAL 0)
84  message(FATAL_ERROR "Build failed with return code ${rv}")
85 endif ()
