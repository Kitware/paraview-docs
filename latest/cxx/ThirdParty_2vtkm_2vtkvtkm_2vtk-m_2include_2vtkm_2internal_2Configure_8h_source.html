228 // Define a pair of macros, VTKM_VECTORIZATION_PRE_LOOP and VTKM_VECTORIZATION_IN_LOOP,
229 // that should be wrapped around any "for"/"while" that you want vectorized.
230 // This is used to set per compiler pragmas for vectorization, and to disable
231 // any warnings that about vectorization failures.
232 /* #undef VTKM_VECTORIZATION_ENABLED */
233 
234 #if defined(VTKM_COMPILER_SUPPORTS_VECTOR_PRAGMAS) &&\
235  defined(VTKM_VECTORIZATION_ENABLED)
236 #if defined(VTKM_CLANG)
237 //clang only needs pre loop
238 #define VTKM_VECTORIZATION_PRE_LOOP \
239  _Pragma("clang loop vectorize(enable) interleave(enable)")
240 #define VTKM_VECTORIZATION_IN_LOOP
241 #elif defined(VTKM_ICC) && defined(NDEBUG)
242 //Note: icc can't do vectorization in debug builds
243 //icc needs pre and in loop
244 //For ICC we want ivdep over simd for the following reason:
245 //#pragma simd is a more powerful combo of '#pragma vector always' and '#pragma ivdep'
246 //The compiler does not check for aliasing or dependencies that might cause
247 //incorrect results after vectorization, and it does not protect against illegal
248 //memory references. #pragma ivdep overrides potential dependencies, but the
249 //compiler still performs a dependency analysis, and will not vectorize if it
250 //finds a proven dependency that would affect results. With #pragma simd, the
251 //compiler does no such analysis, and tries to vectorize regardless.
252 //
253 //Final: We are currently disabling all vectorization with ICC
254 // in the short-term. Both ivdep and simd cause a horrible decrease in compile
255 // time, and generates bad vectorization code.
256 #define VTKM_VECTORIZATION_PRE_LOOP
257 #define VTKM_VECTORIZATION_IN_LOOP
258 #elif defined(VTKM_GCC)
259 //gcc only needs in loop
260 #define VTKM_VECTORIZATION_PRE_LOOP \
261  _Pragma("GCC ivdep")
262 #define VTKM_VECTORIZATION_IN_LOOP
263 #else
264 // Compiler is unknown so we don't define any vectortization pragmas
265 #define VTKM_VECTORIZATION_PRE_LOOP
266 #define VTKM_VECTORIZATION_IN_LOOP
267 #endif
268 #else
269 // Compiler doesn't support any vectortization pragmas
270 #define VTKM_VECTORIZATION_PRE_LOOP
271 #define VTKM_VECTORIZATION_IN_LOOP

325 // Define a pair of macros, VTKM_SWALLOW_SEMICOLON_PRE_BLOCK and
326 // VTKM_SWALLOW_SEMICOLON_POST_BLOCK that can be used around a block in a
327 // macro to "swallow" the semicolon after the macro's use so that it is
328 // used like a function without warnings about extra semicolons. It is
329 // generally implemented by wrapping the block as a do/while with a false
330 // condition so that it executes exactly once. (See, for example,
331 // https://gcc.gnu.org/onlinedocs/cpp/Swallowing-the-Semicolon.html)
332 // However, some compilers might complain about this, too. For example, you
333 // might get a complaint about a loop with a constant expression. These
334 // macros take care of these conditions.
335 #ifdef VTKM_MSVC
336 #define VTKM_SWALLOW_SEMICOLON_PRE_BLOCK \
337  __pragma(warning(push)) \
338  __pragma(warning(disable:4127)) \
339  do
340 #define VTKM_SWALLOW_SEMICOLON_POST_BLOCK \
341  while (false) \
342  __pragma(warning(pop))
343 #else // not VTKM_MSVC
344 #define VTKM_SWALLOW_SEMICOLON_PRE_BLOCK \
345  do
346 #define VTKM_SWALLOW_SEMICOLON_POST_BLOCK \
347  while (false)
348 #endif
