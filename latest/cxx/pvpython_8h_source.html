36 void ProcessArgsForPython(
37  std::vector<char*>& pythonArgs, const char* script, int argc, char* argv[])
38 {
39  pythonArgs.clear();
40  pythonArgs.push_back(vtksys::SystemTools::DuplicateString(argv[0]));
41  if (script)

48  }

56 int Run(int processType, int argc, char* argv[])
57 {
58  // Setup options
59  // Marking this static avoids the false leak messages from vtkDebugLeaks when
60  // using mpich. It appears that the root process which spawns all the
61  // main processes waits in MPI_Init() and calls exit() when
62  // the others are done, causing apparent memory leaks for any non-static objects
63  // created before MPI_Init().
64  vtkInitializationHelper::SetApplicationName("ParaView");
65  static vtkSmartPointer<vtkPVPythonOptions> options = vtkSmartPointer<vtkPVPythonOptions>::New();
66  vtkInitializationHelper::Initialize(argc, argv, processType, options);
67  if (options->GetTellVersion() || options->GetHelpSelected() || options->GetPrintMonitors())
68  {
69  vtkInitializationHelper::Finalize();
70  return 1;
71  }
72 
73  if (processType == vtkProcessModule::PROCESS_BATCH && options->GetPythonScriptName() == 0)
74  {
75  vtkGenericWarningMacro("No script specified. "
76  "Please specify a batch script or use 'pvpython'.");
77  vtkInitializationHelper::Finalize();
78  return 1;
79  }
80 
81  vtkProcessModule* pm = vtkProcessModule::GetProcessModule();
82 
83  // register callback to initialize modules statically. The callback is
84  // empty when BUILD_SHARED_LIBS is ON.
85  vtkPVInitializePythonModules();
86 
87  int ret_val = 0;
88  if (pm->GetSymmetricMPIMode() == false && pm->GetPartitionId() > 0)
89  {
90  vtkIdType sid = vtkSMSession::ConnectToSelf();
91  pm->GetGlobalController()->ProcessRMIs();
92  pm->UnRegisterSession(sid);
93  }
94  else
95  {
96  int remaining_argc;
97  char** remaining_argv;
98  options->GetRemainingArguments(&remaining_argc, &remaining_argv);
99 
100  // Process arguments
101  std::vector<char*> pythonArgs;
102  ProcessArgsForPython(
103  pythonArgs, options->GetPythonScriptName(), remaining_argc, remaining_argv);
104 
105  // Start interpretor
106  vtkPythonInterpreter::Initialize();
107 
108  ret_val =
109  vtkPythonInterpreter::PyMain(static_cast<int>(pythonArgs.size()), &*pythonArgs.begin());
110 
111  // Free python args
112  std::vector<char*>::iterator it = pythonArgs.begin();
113  while (it != pythonArgs.end())
114  {
115  delete[] * it;
116  ++it;
117  }
118  }
119  // Exit application
120  vtkInitializationHelper::Finalize();
121  return ret_val;
122 }
