59  // This version gets slower as the grid increases, while the VTKM version
60  // scales with number of points. This means we can get away with a much finer
61  // grid with the VTKM filter, so we'll just reduce the mesh quality a bit
62  // here.

66 
67  // This is the same equation used in the old implementation:

70  // 1.0 --> 160
71  int divs = static_cast<int>(150 * factor) + 10;
72  this->SetNumberOfDivisions(divs, divs, divs);
73  }
