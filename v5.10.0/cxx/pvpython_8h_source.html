44 void ProcessArgsForPython(std::vector<char*>& pythonArgs, const std::vector<std::string>& args,
45  int vtkNotUsed(argc), char** argv)
46 {
47  pythonArgs.clear();
48 
49  // push the executable name first.
50  pythonArgs.push_back(vtksys::SystemTools::DuplicateString(argv[0]));
51 
52  // now push the unparsed arguments.
53  if (args.empty())
54  {
55  return;
56  }
57 
58  // here we handle a special case when the filename specified is a zip
59  // archive.
60  if (vtksys::SystemTools::GetFilenameLastExtension(args[0]) == ".zip")
61  {
62  // add the archive to sys.path
63  vtkPythonInterpreter::PrependPythonPath(args[0].c_str());
64  pythonArgs.push_back(vtksys::SystemTools::DuplicateString("-m"));
65 
66  std::string modulename = vtksys::SystemTools::GetFilenameWithoutLastExtension(
67  vtksys::SystemTools::GetFilenameName(args[0].c_str()));
68  pythonArgs.push_back(vtksys::SystemTools::DuplicateString(modulename.c_str()));
69  }
70  else
71  {
72  pythonArgs.push_back(vtksys::SystemTools::DuplicateString(args[0].c_str()));
73  }
74 
75  for (size_t cc = 1, max = args.size(); cc < max; ++cc)
76  {
77  pythonArgs.push_back(vtksys::SystemTools::DuplicateString(args[cc].c_str()));
78  }
79 }

82 int Run(int processType, int argc, char* argv[])
83 {
84  // Setup options
85  vtkInitializationHelper::SetApplicationName("ParaView");
86 
87  auto options = vtk::TakeSmartPointer(vtkCLIOptions::New());
88  const auto status = vtkInitializationHelper::Initialize(argc, argv, processType, options);
89  if (!status)
90  {
91  return vtkInitializationHelper::GetExitCode();
92  }
93 
94  if (processType == vtkProcessModule::PROCESS_BATCH && options->GetExtraArguments().empty())
95  {
96  vtkLogF(ERROR, "No script specified. Please specify a batch script or use 'pvpython'.");
97  return EXIT_FAILURE;
98  }
99 
100  vtkProcessModule* pm = vtkProcessModule::GetProcessModule();
101 
102  // register callback to initialize modules statically. The callback is
103  // empty when BUILD_SHARED_LIBS is ON.
104  vtkPVInitializePythonModules();
105 
106  // register static plugins
107  ParaView_paraview_plugins_initialize();
108 
109  vtkPVPluginTracker::GetInstance()->LoadPluginConfigurationXMLs("paraview");
110 
111  int ret_val = 0;
112  if (pm->GetSymmetricMPIMode() == false && pm->GetPartitionId() > 0)
113  {
114  vtkIdType sid = vtkSMSession::ConnectToSelf();
115  pm->GetGlobalController()->ProcessRMIs();
116  pm->UnRegisterSession(sid);
117  }
118  else
119  {
120  // Process arguments
121  std::vector<char*> pythonArgs;
122  ProcessArgsForPython(pythonArgs, options->GetExtraArguments(), argc, argv);
123  pythonArgs.push_back(nullptr);
124 
125  // if user specified verbosity option on command line, then we make vtkPythonInterpreter post
126  // log information as INFO, otherwise we leave it at default which is TRACE.
127  auto pmConfig = vtkProcessModuleConfiguration::GetInstance();
128  vtkPythonInterpreter::SetLogVerbosity(
129  pmConfig->GetLogStdErrVerbosity() != vtkLogger::VERBOSITY_INVALID
130  ? vtkLogger::VERBOSITY_INFO
131  : vtkLogger::VERBOSITY_TRACE);
132 
133  // Start interpretor
134  vtkPythonInterpreter::Initialize();
135 
136  ret_val =
137  vtkPythonInterpreter::PyMain(static_cast<int>(pythonArgs.size()) - 1, &pythonArgs.front());
138 
139  // Free python args
140  for (auto& ptr : pythonArgs)
141  {
142  delete[] ptr;
143  }
144  }
145  // Exit application
146  vtkInitializationHelper::Finalize();
147  return ret_val;
148 }
