50 inline std::int32_t getFirstVisibleBlockId(const QTextEdit& text)
51 {
52  QTextCursor curs = QTextCursor(text.document());
53  curs.movePosition(QTextCursor::Start);
54  for (std::int32_t i = 0; i < text.document()->blockCount(); ++i)
55  {
56  QTextBlock block = curs.block();
57 
58  QRect r1 = text.viewport()->geometry();
59  QRect r2 = text.document()
60  ->documentLayout()
61  ->blockBoundingRect(block)
62  .translated(text.viewport()->geometry().x(),
63  text.viewport()->geometry().y() - (text.verticalScrollBar()->sliderPosition()))
64  .toRect();
65 
66  if (r1.contains(r2, true))
67  {
68  return i;
69  }
70 
71  curs.movePosition(QTextCursor::NextBlock);
72  }
73 
74  return 0;
75 };
