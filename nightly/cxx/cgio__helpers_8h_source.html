41 //------------------------------------------------------------------------------
42 template <typename T>
43 inline int readNodeData(int cgioNum, double nodeId, std::vector<T>& data)
44 {
45  int n;
46  cgsize_t size = 1;
47  cgsize_t dimVals[12];
48  int ndim;
49 
50  if (cgio_get_dimensions(cgioNum, nodeId, &ndim, dimVals) != CG_OK)
51  {
52  cgio_error_exit("cgio_get_dimensions");
53  return 1;
54  }
55 
56  // allocate data
57  for (n = 0; n < ndim; n++)
58  {
59  size *= dimVals[n];
60  }
61  if (size <= 0)
62  {
63  return 1;
64  }
65  data.resize(size);
66 
67  // read data
68  if (cgio_read_all_data(cgioNum, nodeId, &data[0]) != CG_OK)
69  {
70  return 1;
71  }
72 
73  return 0;
74 }

76 //------------------------------------------------------------------------------
77 /*
78  * Converts data read from the file using native type to the type specified
79  * as the template argument. Just uses static_cast to do type conversion.
80  */
81 template <typename T>
82 inline int readNodeDataAs(int cgioNum, double nodeId, std::vector<T>& data)
83 {
84  // let's get type in file.
85  char dtype[CGIO_MAX_DATATYPE_LENGTH + 1];
86  if (cgio_get_data_type(cgioNum, nodeId, dtype) != CG_OK)
87  {
88  cgio_error_exit("cgio_get_data_type");
89  return 1;
90  }
91 
92  if (strcmp(dtype, "I4") == 0)
93  {
94  std::vector<vtkTypeInt32> i32vector;
95  readNodeData<vtkTypeInt32>(cgioNum, nodeId, i32vector);
96  data.resize(i32vector.size());
97  std::copy(i32vector.begin(), i32vector.end(), data.begin());
98  }
99  else if (strcmp(dtype, "I8") == 0)
100  {
101  std::vector<vtkTypeInt64> i64vector;
102  readNodeData<vtkTypeInt64>(cgioNum, nodeId, i64vector);
103  data.resize(i64vector.size());
104  std::copy(i64vector.begin(), i64vector.end(), data.begin());
105  }
106  else if (strcmp(dtype, "R4") == 0)
107  {
108  std::vector<float> fvector;
109  readNodeData<float>(cgioNum, nodeId, fvector);
110  data.resize(fvector.size());
111  std::copy(fvector.begin(), fvector.end(), data.begin());
112  }
113  else if (strcmp(dtype, "R8") == 0)
114  {
115  std::vector<double> dvector;
116  readNodeData<double>(cgioNum, nodeId, dvector);
117  data.resize(dvector.size());
118  std::copy(dvector.begin(), dvector.end(), data.begin());
119  }
120  else
121  {
122  return 1;
123  }
124  return CG_OK;
125 }
