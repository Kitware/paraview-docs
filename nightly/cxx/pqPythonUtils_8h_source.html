47 inline std::int32_t GetFirstVisibleBlockId(const QTextEdit& text)
48 {
49  QTextCursor curs = QTextCursor(text.document());
50  curs.movePosition(QTextCursor::Start);
51  for (std::int32_t i = 0; i < text.document()->blockCount(); ++i)
52  {
53  QTextBlock block = curs.block();
54 
55  QRect r1 = text.viewport()->geometry();
56  QRect r2 = text.document()
57  ->documentLayout()
58  ->blockBoundingRect(block)
59  .translated(text.viewport()->geometry().x(),
60  text.viewport()->geometry().y() - (text.verticalScrollBar()->sliderPosition()))
61  .toRect();
62 
63  if (r1.contains(r2, true))
64  {
65  return i;
66  }
67 
68  curs.movePosition(QTextCursor::NextBlock);
69  }
70 
71  return 0;
72 };
