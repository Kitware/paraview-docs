29 template <typename FunctorInternal>
30 void vtkSMPTools_Impl_For(vtkIdType first, vtkIdType last, vtkIdType grain, FunctorInternal& fi)
31 {
32  vtkIdType n = last - first;
33  if (!n)
34  {
35  return;
36  }
37 
38  if (grain == 0 || grain >= n)
39  {
40  fi.Execute(first, last);
41  }
42  else
43  {
44  vtkIdType b = first;
45  while (b < last)
46  {
47  vtkIdType e = b + grain;
48  if (e > last)
49  {
50  e = last;
51  }
52  fi.Execute(b, e);
53  b = e;
54  }
55  }
56 }
