5 FindMEMKIND
6 ---------
7 
8 Find memkind's headers and library.
9 
10 Imported Targets
11 ^^^^^^^^^^^^^^^^
12 
13 This module defines the following :prop_tgt:`IMPORTED` targets:
14 
15 ``MEMKIND::MEMKIND``
16  The memkind library, if found.
17 
18 Result Variables
19 ^^^^^^^^^^^^^^^^
20 
21 This module will set the following variables in your project:
22 
23 ``MEMKIND_INCLUDE_DIRS``
24  where to find memkind.h, etc.
25 ``MEMKIND_LIBRARIES``
26  the libraries to link against to use memkind.
27 ``MEMKIND_FOUND``
28  true if the memkind headers and libraries were found.
29 
30 #]=======================================================================]
31 include(CMakeFindDependencyMacro)
32 
33 # Look for the header file.
34 find_path(MEMKIND_INCLUDE_DIR
35  NAMES memkind.h
36  DOC "memkind include directory")
37 mark_as_advanced(MEMKIND_INCLUDE_DIR)
38 
39 # Look for the library.
40 find_library(MEMKIND_LIBRARY
41  NAMES memkind libmemkind
42  DOC "memkind library")
43 mark_as_advanced(MEMKIND_LIBRARY)
44 
45 include(FindPackageHandleStandardArgs)
46 find_package_handle_standard_args(MEMKIND
47  REQUIRED_VARS MEMKIND_LIBRARY MEMKIND_INCLUDE_DIR)
48 
49 # Copy the results to the output variables and target.
50 if(MEMKIND_FOUND)
51  set(MEMKIND_LIBRARIES "${MEMKIND_LIBRARY}")
52  set(MEMKIND_INCLUDE_DIRS "${MEMKIND_INCLUDE_DIR}")
53 
54  if(NOT TARGET MEMKIND::MEMKIND)
55  add_library(MEMKIND::MEMKIND UNKNOWN IMPORTED)
56  set_target_properties(MEMKIND::MEMKIND PROPERTIES
57  IMPORTED_LINK_INTERFACE_LANGUAGES "C"
58  IMPORTED_LOCATION "${MEMKIND_LIBRARY}"
59  INTERFACE_INCLUDE_DIRECTORIES "${MEMKIND_INCLUDE_DIRS}")
60  endif()
61 endif()
