[docs]def update_camera(viewProxy, cameraData):
    viewProxy.CameraFocalPoint = cameraData["focalPoint"]
    viewProxy.CameraPosition = cameraData["position"]
    viewProxy.CameraViewUp = cameraData["viewUp"]
    simple.Render(viewProxy)
