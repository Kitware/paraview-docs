34 static bool RealMain(int argc, char* argv[], vtkProcessModule::ProcessTypes type)
35 {
36  // Marking this static avoids the false leak messages from vtkDebugLeaks when
37  // using mpich. It appears that the root process which spawns all the
38  // main processes waits in MPI_Init() and calls exit() when
39  // the others are done, causing apparent memory leaks for any non-static objects
40  // created before MPI_Init().
41  static vtkSmartPointer<vtkPVServerOptions> options = vtkSmartPointer<vtkPVServerOptions>::New();
42 
43  // Init current process type
44  vtkInitializationHelper::Initialize(argc, argv, type, options);
45  if (options->GetTellVersion() || options->GetHelpSelected() || options->GetPrintMonitors())
46  {
47  vtkInitializationHelper::Finalize();
48  return 1;
49  }
50 
51 #ifdef PARAVIEW_ENABLE_PYTHON
52  // register callback to initialize modules statically. The callback is
53  // empty when BUILD_SHARED_LIBS is ON.
54  vtkPVInitializePythonModules();
55 #endif
56 
57 #ifndef BUILD_SHARED_LIBS
58  // load static plugins
59  paraview_static_plugins_init();
60 #endif
61 
62  vtkProcessModule* pm = vtkProcessModule::GetProcessModule();
63  vtkMultiProcessController* controller = pm->GetGlobalController();
64 
65  vtkPVSessionServer* session = vtkPVSessionServer::New();
66  session->SetMultipleConnection(options->GetMultiClientMode() != 0);
67  session->SetDisableFurtherConnections(options->GetDisableFurtherConnections() != 0);
68 
69  int process_id = controller->GetLocalProcessId();
70  if (process_id == 0)
71  {
72  // Report status:
73  if (options->GetReverseConnection())
74  {
75  cout << "Connecting to client (reverse connection requested)..." << endl;
76  }
77  else
78  {
79  cout << "Waiting for client..." << endl;
80  }
81  }
82  bool success = false;
83  if (session->Connect())
84  {
85  success = true;
86  pm->RegisterSession(session);
87  if (controller->GetLocalProcessId() == 0)
88  {
89  while (pm->GetNetworkAccessManager()->ProcessEvents(0) != -1)
90  {
91  }
92  }
93  else
94  {
95  controller->ProcessRMIs();
96  }
97  pm->UnRegisterSession(session);
98  }
99 
100  cout << "Exiting..." << endl;
101  session->Delete();
102  // Exit application
103  vtkInitializationHelper::Finalize();
104  return success;
105 }
