156  //---------------------------------------------------------------------------
157  // Makes is easy to obtain a value for a series parameter, is set, else the
158  // default. This class supports two mechanisms for addresses series in a
159  // collection (multiblock) of tables: (1) using a name that combines the
160  // table name and the column name (using
161  // vtkChartRepresentation::GetDefaultSeriesLabel), or (2) using the column
162  // name alone. (1) is always checked before (2).
163  template <class T>
164  T GetSeriesParameter(vtkXYChartRepresentation* self, const std::string& tableName,
165  const std::string& columnName, const std::string& vtkNotUsed(role),
166  const std::map<std::string, T>& parameter_map, const T default_value = T()) const
167  {
168  typename std::map<std::string, T>::const_iterator iter;
169 
170  // when setting properties for a series, I want to support two mechanisms:
171  // simply specifying the array name or suffixing it with the block-name.
172  // This logic makes that possible.
173 
174  // first try most specific form of identifying the series.
175  std::string key = self->GetDefaultSeriesLabel(tableName, columnName);
176  iter = parameter_map.find(key);
177  if (iter != parameter_map.end())
178  {
179  return iter->second;
180  }
181 
182  // now try the cheap form for identifying it.
183  key = self->GetDefaultSeriesLabel(std::string(), columnName);
184  iter = parameter_map.find(key);
185  if (iter != parameter_map.end())
186  {
187  return iter->second;
188  }
189  return default_value;
190  }

262  //---------------------------------------------------------------------------
263  // Update i.e. add/remove plots based on the data in the tables.
264  virtual void UpdatePlots(vtkXYChartRepresentation* self, const MapOfTables& tables)
265  {
266  PlotsMap newPlots;
267  assert(self != NULL);
268  vtkChartXY* chartXY = self->GetChart();
269  this->RemoveAllPlots(chartXY);
270  std::multimap<int, std::pair<vtkTable*, std::array<std::string, 3> > > orderMap;
271  for (MapOfTables::const_iterator tablesIter = tables.begin(); tablesIter != tables.end();
272  ++tablesIter)
273  {
274  const std::string& tableName = tablesIter->first;
275  vtkTable* table = tablesIter->second.GetPointer();
276  vtkIdType numCols = table->GetNumberOfColumns();
277  for (vtkIdType cc = 0; cc < numCols; ++cc)
278  {
279  std::string columnName = table->GetColumnName(cc);
280  auto roles = this->GetSeriesRoles(tableName, columnName);
281 
282  for (const auto& role : roles)
283  {
284  // recover the order of the current table column
285  const int order =
286  this->GetSeriesParameter(self, tableName, columnName, role, this->SeriesOrder, -1);
287 
288  // store all info in a (sorted) map
289  std::array<std::string, 3> mapValue;
290  mapValue[0] = tableName;
291  mapValue[1] = columnName;
292  mapValue[2] = role;
293  orderMap.insert(std::make_pair(order, std::make_pair(table, mapValue)));
294  }
295  }
296  }
297 
298  // for each ordered column
299  for (auto it = orderMap.begin(); it != orderMap.end(); ++it)
300  {
301  vtkTable* table = it->second.first;
302  const std::string& tableName = it->second.second[0];
303  const std::string& columnName = it->second.second[1];
304  const std::string& role = it->second.second[2];
305 
306  vtkSmartPointer<vtkPlot> plot = this->SeriesPlots.GetPlot(self, tableName, columnName, role);
307  if (!plot)
308  {
309  // Create the plot in the right order
310  plot = this->NewPlot(self, tableName, columnName, role);
311  if (!plot)
312  {
313  continue;
314  }
315  }
316  plot->SetInputData(table);
317  plot->SetUseIndexForXSeries(self->GetUseIndexForXAxis());
318  plot->SetInputArray(0, self->GetXAxisSeriesName());
319  plot->SetInputArray(this->GetInputArrayIndex(tableName, columnName, role), columnName);
320  this->SeriesPlots.AddPlot(self, tableName, columnName, role, plot);
321  newPlots.AddPlot(self, tableName, columnName, role, plot);
322  }
323 
324  // Remove any plots in this->SeriesPlots that are not in newPlots.
325  this->SeriesPlots.Intersect(newPlots, chartXY);
326  }

329  // Update properties for plots in the chart.
330  virtual void UpdatePlotProperties(vtkXYChartRepresentation* self)
331  {
332  vtkChartXY* chartXY = self->GetChart();
333  vtkPlot* lastFunctionalBagPlot = 0;
334  for (PlotsMap::iterator iter1 = this->SeriesPlots.begin(); iter1 != this->SeriesPlots.end();
335  ++iter1)
336  {
337  for (PlotsMapItem::const_iterator iter2 = iter1->second.begin(); iter2 != iter1->second.end();
338  ++iter2)
339  {
340  const PlotInfo& plotInfo = iter2->second;
341  const std::string& tableName = plotInfo.TableName;
342  const std::string& columnName = plotInfo.ColumnName;
343  vtkPlot* plot = plotInfo.Plot;
344  const std::string& role = iter2->first;
345 
346  if (this->UpdateSinglePlotProperties(self, tableName, columnName, role, plot))
347  {
348  // Functional bag plots shall be stacked under the other plots.
349  vtkPlotFunctionalBag* plotBag = vtkPlotFunctionalBag::SafeDownCast(plot);
350  if (plotBag)
351  {
352  // We can't select the median line as it may not exist in other dataset.
353  if (columnName == "QMedianLine")
354  {
355  plotBag->SelectableOff();
356  }
357  if (plotBag->IsBag())
358  {
359  if (!lastFunctionalBagPlot)
360  {
361  chartXY->LowerPlot(plotBag);
362  }
363  else
364  {
365  chartXY->StackPlotAbove(plotBag, lastFunctionalBagPlot);
366  }
367  lastFunctionalBagPlot = plotBag;
368  }
369  }
370  }
371  }

409  virtual bool UpdateSinglePlotProperties(vtkXYChartRepresentation* self,

433  plot->SetColor(color.GetRed(), color.GetGreen(), color.GetBlue());
434  plot->GetSelectionPen()->SetColorF(self->SelectionColor);
435 
436  plot->SetWidth(
437  this->GetSeriesParameter(self, tableName, columnName, role, this->LineThicknesses, 2));
438  plot->GetPen()->SetLineType(this->GetSeriesParameter(
439  self, tableName, columnName, role, this->LineStyles, static_cast<int>(vtkPen::SOLID_LINE)));
440 
441  if (vtkPlotPoints* plotPoints = vtkPlotPoints::SafeDownCast(plot))
442  {
443  plotPoints->SetMarkerStyle(this->GetSeriesParameter(self, tableName, columnName, role,
444  this->MarkerStyles, static_cast<int>(vtkPlotPoints::NONE)));
445  // the vtkValidPointMask array is used by some filters (like plot
446  // over line) to indicate invalid points. this instructs the line
447  // plot to not render those points
448  plotPoints->SetValidPointMaskName("vtkValidPointMask");
449  }
450 
451  chartXY->SetPlotCorner(
452  plot, this->GetSeriesParameter(self, tableName, columnName, role, this->AxisCorners, 0));
453 
454  // for now only vtkPlotBar has color mapping
455  vtkPlotBar* plotBar = vtkPlotBar::SafeDownCast(plot);
456  if (plotBar && columnName == "bin_values")
457  {
458  bool colorMapping =
459  this->GetSeriesParameter(self, tableName, columnName, role, this->UseColorMapping, false);
460  plotBar->SetScalarVisibility(colorMapping);
461  plotBar->SelectColorArray("bin_extents");
462  vtkScalarsToColors* lut = this->GetSeriesParameter(
463  self, tableName, columnName, role, this->Lut, static_cast<vtkScalarsToColors*>(NULL));
464  if (lut)
465  {
466  plotBar->SetLookupTable(lut);
467  }
468  }
469  return true;
470  }
