180  {
181  bool modified = false;
182  unsigned int numArgs = this->GetNumberOfUncheckedElements();
183  if (numArgs != numValues)
184  {
185  this->UncheckedValues.resize(numValues);
186  numArgs = numValues;
187  modified = true;
188  }

190  {
191  modified = !std::equal(this->UncheckedValues.begin(), this->UncheckedValues.end(), values);
192  }

195  {
196  return 1;
197  }
198

205  //---------------------------------------------------------------------------
206  int SetElement(unsigned int idx, T value)
207  {
208  unsigned int numElems = this->GetNumberOfElements();
209 
210  if (this->Initialized && idx < numElems && value == this->GetElement(idx))
211  {
212  return 1;
213  }
214 
215  if (idx >= numElems)
216  {
217  this->SetNumberOfElements(idx + 1);
218  }
219  this->Values[idx] = value;
220 
221  // Make sure to initialize BEFORE Modified() is called. Otherwise,
222  // the value would not be pushed.
223  this->Initialized = true;
224  this->Property->Modified();
225  this->ClearUncheckedElements();
226  return 1;
227  }

235  //---------------------------------------------------------------------------
236  int SetElements(const T* values, unsigned int numValues)
237  {
238  bool modified = false;
239  unsigned int numArgs = this->GetNumberOfElements();
240  if (numArgs != numValues)
241  {
242  this->Values.resize(numValues);
243  this->UncheckedValues.resize(numValues);
244  numArgs = numValues;
245  modified = true;
246  }
247  else
248  {
249  modified = !std::equal(this->Values.begin(), this->Values.end(), values);
250  }
251  if (!modified && this->Initialized)
252  {
253  return 1;
254  }
255 
256  std::copy(values, values + numArgs, this->Values.begin());
257  this->Initialized = true;
258  if (!modified && numValues == 0)
259  {
260  // handle the case when the property didn't have valid values but the new
261  // values don't really change anything. In that case, the property hasn't
262  // really been modified, so skip invoking the event. This keeps Python
263  // trace from ending up with lots of properties such as EdgeBlocks etc for
264  // ExodusIIReader which haven't really changed at all.
265  }
266  else
267  {
268  this->Property->Modified();
269  this->ClearUncheckedElements();
270  }
271  return 1;
272  }

276  {

280

293  this->Property->Modified();
294  }

298  {
299  this->UncheckedValues = dsrc->Values;
300  modified = true;

331  bool LoadStateValues(vtkPVXMLElement* element)
332  {
333  if (!element)
334  {
335  return false;
336  }
337 
338  std::vector<T> new_values;
339  unsigned int numElems = element->GetNumberOfNestedElements();
340  for (unsigned int i = 0; i < numElems; i++)
341  {
342  vtkPVXMLElement* current = element->GetNestedElement(i);
343  if (current->GetName() && strcmp(current->GetName(), "Element") == 0)
344  {
345  int index;
346  const char* str_value = current->GetAttribute("value");
347  if (str_value && current->GetScalarAttribute("index", &index) && index >= 0)
348  {
349  if (index <= static_cast<int>(new_values.size()))
350  {
351  new_values.resize(index + 1);
352  }
353 
354  new_values[index] = vtkSMVPConvertFromString<T>(str_value);
355  }
356  }
357  }
358  if (new_values.size() > 0)
359  {
360  this->SetElements(&new_values[0], static_cast<unsigned int>(new_values.size()));
361  }
362  else
363  {
364  this->SetNumberOfElements(0);
365  }
366 
367  return true;
368  }

372  {
373  unsigned int size = this->GetNumberOfElements();
374  if (size > 0)
375  {
376  propertyElement->AddAttribute("number_of_elements", size);
377  }
378 
379  // helps save full precision doubles and floats.
380  vtkNumberToString convertor;

384  stream << convertor(this->GetElement(i));
385 
386  vtkPVXMLElement* elementElement = vtkPVXMLElement::New();
387  elementElement->SetName("Element");
388  elementElement->AddAttribute("index", i);
389  elementElement->AddAttribute("value", stream.str().c_str());
390  propertyElement->AddNestedElement(elementElement);
391  elementElement->Delete();
392  }
