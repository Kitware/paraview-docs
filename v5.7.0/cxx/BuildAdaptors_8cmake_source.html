42 # Function to easy adding separate custom-commands to build the adaptors.
43 #------------------------------------------------------------------------------
44 function(build_adaptor name languages)
45  string(TOLOWER "${name}" lname)
46 
47  set(language_options)
48  foreach (lang IN LISTS languages)
49  list(APPEND language_options
50  -DCMAKE_${lang}_COMPILER:FILEPATH=${CMAKE_${lang}_COMPILER}
51  -DCMAKE_${lang}_FLAGS:STRING=${CMAKE_${lang}_FLAGS})
52  endforeach ()
53 
54  add_custom_command(
55  OUTPUT "${BINARY_DIR}/${lname}.done"
56  COMMAND ${CMAKE_CTEST_COMMAND}
57  ${build_config_arg}
58  --build-and-test ${SOURCE_DIR}/${name}
59  ${BINARY_DIR}/${name}
60  --build-noclean
61  --build-two-config
62  --build-project ${name}
63  --build-generator ${CMAKE_GENERATOR}
64  --build-makeprogram ${CMAKE_MAKE_PROGRAM}
65  --build-options -DParaView_DIR:PATH=${CMAKE_BINARY_DIR}/${paraview_cmake_destination}
66  -DQT_QMAKE_EXECUTABLE:FILEPATH=${QT_QMAKE_EXECUTABLE}
67  -DQt5_DIR:PATH=${Qt5_DIR}
68  -DCMAKE_BUILD_TYPE:STRING=${CMAKE_BUILD_TYPE}
69  -DCMAKE_CXX_COMPILER:FILEPATH=${CMAKE_CXX_COMPILER}
70  -DCMAKE_CXX_FLAGS:STRING=${CMAKE_CXX_FLAGS}
71  -DCMAKE_LIBRARY_OUTPUT_DIRECTORY:PATH=${CMAKE_LIBRARY_OUTPUT_DIRECTORY}
72  -DCMAKE_RUNTIME_OUTPUT_DIRECTORY:PATH=${CMAKE_RUNTIME_OUTPUT_DIRECTORY}
73  -DCMAKE_PREFIX_PATH:STRING="${CMAKE_PREFIX_PATH}"
74  ${language_options}
75  ${extra_params}
76  --no-warn-unused-cli
77  COMMAND ${CMAKE_COMMAND}
78  -E touch "${BINARY_DIR}/${lname}.done"
79 
80  ${ARGN}
81  )
82  add_custom_target(${name} ALL DEPENDS "${BINARY_DIR}/${lname}.done")
83 endfunction()
