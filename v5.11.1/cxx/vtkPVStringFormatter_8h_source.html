371  template <typename T>
372  void AddArg(const fmt::detail::named_arg<char_type, T>& fmtArg)
373  {
374  bool argNotFound = std::find_if(this->Arguments.begin(), this->Arguments.end(),
375  [&fmtArg](const vtkNamedArgument& arg) {
376  return arg.Name == fmtArg.name;
377  }) == this->Arguments.end();
378  // if argument was not found
379  if (argNotFound)
380  {
381  vtkNamedArgument newArg(fmtArg.name, fmtArg.value);
382  this->Arguments.push_back(newArg);
383  }
384  else // else print warning
385  {
386  vtkLogF(WARNING, "Argument %s already exists. Try to add another one.", fmtArg.name);
387  }
