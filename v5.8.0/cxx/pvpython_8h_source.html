38 void ProcessArgsForPython(
39  std::vector<char*>& pythonArgs, const char* script, int argc, char* argv[])
40 {
41  pythonArgs.clear();
42  pythonArgs.push_back(vtksys::SystemTools::DuplicateString(argv[0]));
43  if (script)
44  {
45  pythonArgs.push_back(vtksys::SystemTools::DuplicateString(script));
46  }
47  else if (argc > 1)
48  {
49  pythonArgs.push_back(vtksys::SystemTools::DuplicateString("-"));
50  }
51  for (int cc = 1; cc < argc; cc++)
52  {
53  pythonArgs.push_back(vtksys::SystemTools::DuplicateString(argv[cc]));
54  }
55 }

58 int Run(int processType, int argc, char* argv[])
59 {
60  // Setup options
61  // Marking this static avoids the false leak messages from vtkDebugLeaks when
62  // using mpich. It appears that the root process which spawns all the
63  // main processes waits in MPI_Init() and calls exit() when
64  // the others are done, causing apparent memory leaks for any non-static objects
65  // created before MPI_Init().
66  vtkInitializationHelper::SetApplicationName("ParaView");
67  static vtkSmartPointer<vtkPVPythonOptions> options = vtkSmartPointer<vtkPVPythonOptions>::New();
68  vtkInitializationHelper::Initialize(argc, argv, processType, options);
69  if (options->GetTellVersion() || options->GetHelpSelected() || options->GetPrintMonitors())
70  {
71  vtkInitializationHelper::Finalize();
72  return EXIT_SUCCESS;
73  }
74 
75  if (processType == vtkProcessModule::PROCESS_BATCH && options->GetPythonScriptName() == 0)
76  {
77  vtkGenericWarningMacro("No script specified. "
78  "Please specify a batch script or use 'pvpython'.");
79  vtkInitializationHelper::Finalize();
80  return EXIT_FAILURE;
81  }
82 
83  vtkProcessModule* pm = vtkProcessModule::GetProcessModule();
84 
85  // register callback to initialize modules statically. The callback is
86  // empty when BUILD_SHARED_LIBS is ON.
87  vtkPVInitializePythonModules();
88 
89  vtkPVPluginTracker::GetInstance()->LoadPluginConfigurationXMLs("paraview");
90 
91  int ret_val = 0;
92  if (pm->GetSymmetricMPIMode() == false && pm->GetPartitionId() > 0)
93  {
94  vtkIdType sid = vtkSMSession::ConnectToSelf();
95  pm->GetGlobalController()->ProcessRMIs();
96  pm->UnRegisterSession(sid);
97  }
98  else
99  {
100  int remaining_argc;
101  char** remaining_argv;
102  options->GetRemainingArguments(&remaining_argc, &remaining_argv);
103 
104  // Process arguments
105  std::vector<char*> pythonArgs;
106  ProcessArgsForPython(
107  pythonArgs, options->GetPythonScriptName(), remaining_argc, remaining_argv);
108  pythonArgs.push_back(nullptr);
109 
110  // if user specified verbosity option on command line, then we make vtkPythonInterpreter post
111  // log information as INFO, otherwise we leave it at default which is TRACE.
112  vtkPythonInterpreter::SetLogVerbosity(
113  options->GetLogStdErrVerbosity() != vtkLogger::VERBOSITY_INVALID
114  ? vtkLogger::VERBOSITY_INFO
115  : vtkLogger::VERBOSITY_TRACE);
116 
117  // Start interpretor
118  vtkPythonInterpreter::Initialize();
119 
120  ret_val =
121  vtkPythonInterpreter::PyMain(static_cast<int>(pythonArgs.size()) - 1, &*pythonArgs.begin());
122 
123  // Free python args
124  std::vector<char*>::iterator it = pythonArgs.begin();
125  while (it != pythonArgs.end())
126  {
127  delete[] * it;
128  ++it;
129  }
130  }
131  // Exit application
132  vtkInitializationHelper::Finalize();
133  return ret_val;
134 }
135 }
