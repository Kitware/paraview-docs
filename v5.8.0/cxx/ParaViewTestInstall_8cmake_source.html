33 # This script tests the ParaView install tree by building the examples
34 #
35 # The script expects the following input variables:
36 # PARAVIEW_BINARY_DIR : Build path for ParaView (To run 'make install' here)
37 # PARAVIEW_INSTALL_DIR : Install path for ParaView Examples are built against
38 # (This should be CMAKE_INSTALL_PREFIX set for ParaView)
39 # PARAVIEW_SOURCE_DIR : ParaView source dir (For source location of Examples)
40 # PARAVIEW_TEST_DIR : Temporary directory for location of Examples build tree
41 # PARAVIEW_VERSION : ParaView version string used when creating the installtree
42 
43 message (STATUS "Building Examples against ParaView install tree")
44 message("CTEST_FULL_OUTPUT") # Don't truncate test output.
45 # Remove the drive letter from `PARAVIEW_INSTALL_DIR` so we can append it to
46 # DESTDIR safely.
47 if (WIN32 AND IS_ABSOLUTE "${PARAVIEW_INSTALL_DIR}")
48  string(REGEX REPLACE "^.:" "" PARAVIEW_INSTALL_DIR "${PARAVIEW_INSTALL_DIR}")
49 endif ()
50 set (ParaView_DIR
51  $ENV{DESTDIR}${PARAVIEW_INSTALL_DIR}/${PARAVIEW_CMAKE_DESTINATION})
52 message(STATUS "ParaView_DIR: ${ParaView_DIR}")
53 
54 # Build target "INSTALL" for paraview
55 if (MSVC)
56  set (install_tgt "INSTALL")
57  set (opt "/maxcpucount")
58 else ()
59  set (install_tgt "install")
60  set (opt "-j5")
61 endif ()
62 execute_process (COMMAND ${CMAKE_COMMAND}
63  --build ${PARAVIEW_BINARY_DIR}
64  --target ${install_tgt}
65  -- ${opt}
66  WORKING_DIRECTORY ${PARAVIEW_BINARY_DIR}
67  RESULT_VARIABLE irv)
68 if (NOT irv EQUAL 0)
69  message(FATAL_ERROR "Could not build target 'install' for ParaView")
70 endif ()
71 
72 set(generator_args)
73 if (CMAKE_GENERATOR)
74  list(APPEND generator_args
75  -G "${CMAKE_GENERATOR}")
76 endif ()
77 if (CMAKE_GENERATOR_PLATFORM)
78  list(APPEND generator_args
79  -A "${CMAKE_GENERATOR_PLATFORM}")
80 endif ()
81 if (CMAKE_GENERATOR_TOOLSET)
82  list(APPEND generator_args
83  -T "${CMAKE_GENERATOR_TOOLSET}")
84 endif ()
85 
86 set (INSTALL_TEST_BUILD_DIR ${PARAVIEW_TEST_DIR}/Examples-bld)
87 if (NOT EXISTS ${INSTALL_TEST_BUILD_DIR})
88  file(MAKE_DIRECTORY
89  ${INSTALL_TEST_BUILD_DIR})
90 endif ()
91 execute_process (
92  COMMAND ${CMAKE_COMMAND}
93  ${generator_args}
94  -DParaView_DIR:PATH=${ParaView_DIR}
95  -DBUILD_SHARED_LIBS:BOOL=${BUILD_SHARED_LIBS}
96  ${PARAVIEW_SOURCE_DIR}/Examples
97  WORKING_DIRECTORY ${INSTALL_TEST_BUILD_DIR}
98  RESULT_VARIABLE crv)
99 if (NOT crv EQUAL 0)
100  message(FATAL_ERROR "Configuration failed with return code ${crv}")
101 endif ()
102 execute_process (
103  COMMAND ${CMAKE_COMMAND} --build ${INSTALL_TEST_BUILD_DIR} --clean-first
104  WORKING_DIRECTORY ${INSTALL_TEST_BUILD_DIR}
105  RESULT_VARIABLE rv)
106 if (NOT rv EQUAL 0)
107  message(FATAL_ERROR "Build failed with return code ${rv}")
108 endif ()
