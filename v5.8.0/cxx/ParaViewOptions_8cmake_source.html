222 #[==[
223 Conditionally require/reject optional modules
224 
225 Use this macro to conditionally require (or reject) modules.
226 
227 ~~~
228 paraview_require_module(
229  MODULES <module>...
230  [CONDITION <condition>]
231  [EXCLUSIVE]
232 ~~~
233 
234 The arguments are as follows:
235 
236  * `MODULES`: (Required) The list of modules.
237  * `CONDITION`: (Defaults to `TRUE`) The condition under which the modules
238  specified are added to the requested list.
239  * `EXCLUSIVE`: When sepcified, if `CONDITION` is false, the module will be
240  added to the rejected modules list.
241 #]==]
242 macro (paraview_require_module)
243  cmake_parse_arguments(pem
244  "EXCLUSIVE"
245  ""
246  "CONDITION;MODULES"
247  ${ARGN})
248 
249  if (pem_UNPARSED_ARGUMENTS)
250  message(FATAL_ERROR
251  "Unparsed arguments for `paraview_require_module`: "
252  "${pem_UNPARSED_ARGUMENTS}")
253  endif ()
254 
255  if (NOT DEFINED pem_CONDITION)
256  set(pem_CONDITION TRUE)
257  endif ()
258 
259  if (${pem_CONDITION})
260  # message("${pem_CONDITION} == TRUE")
261  list(APPEND paraview_requested_modules ${pem_MODULES})
262  elseif (pem_EXCLUSIVE)
263  # message("${pem_CONDITION} == FALSE")
264  list(APPEND paraview_rejected_modules ${pem_MODULES})
265  endif()
266  unset(pem_EXCLUSIVE)
267  unset(pem_CONDITION)
268  unset(pem_MODULES)
269  unset(pem_UNPARSED_ARGUMENTS)
270 endmacro()
