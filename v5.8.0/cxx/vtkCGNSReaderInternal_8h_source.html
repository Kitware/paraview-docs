402 template <typename T, typename Y>
403 int get_XYZ_mesh(const int cgioNum, const std::vector<double>& gridChildId,
404  const std::size_t& nCoordsArray, const int cellDim, const vtkIdType nPts,
405  const cgsize_t* srcStart, const cgsize_t* srcEnd, const cgsize_t* srcStride,
406  const cgsize_t* memStart, const cgsize_t* memEnd, const cgsize_t* memStride,
407  const cgsize_t* memDims, vtkPoints* points)
408 {
409  T* coords = static_cast<T*>(points->GetVoidPointer(0));
410  T* currentCoord = static_cast<T*>(&(coords[0]));
411 
412  CGNSRead::char_33 coordName;
413  std::size_t len;
414  bool sameType = true;
415  double coordId;
416 
417  memset(coords, 0, 3 * nPts * sizeof(T));
418 
419  for (std::size_t c = 1; c <= nCoordsArray; ++c)
420  {
421  // Read CoordName
422  if (cgio_get_name(cgioNum, gridChildId[c - 1], coordName) != CG_OK)
423  {
424  char message[81];
425  cgio_error_message(message);
426  std::cerr << "get_XYZ_mesh : cgio_get_name :" << message;
427  }
428 
429  // Read node data type
430  CGNSRead::char_33 dataType;
431  if (cgio_get_data_type(cgioNum, gridChildId[c - 1], dataType))
432  {
433  continue;
434  }
435 
436  if (strcmp(dataType, "R8") == 0)
437  {
438  const bool doubleType = detail::is_double<T>::value;
439  sameType = doubleType;
440  }
441  else if (strcmp(dataType, "R4") == 0)
442  {
443  const bool floatType = detail::is_float<T>::value;
444  sameType = floatType;
445  }
446  else
447  {
448  std::cerr << "Invalid datatype for GridCoordinates\n";
449  continue;
450  }
451 
452  // Determine direction X,Y,Z
453  len = strlen(coordName) - 1;
454  switch (coordName[len])
455  {
456  case 'X':
457  currentCoord = static_cast<T*>(&(coords[0]));
458  break;
459  case 'Y':
460  currentCoord = static_cast<T*>(&(coords[1]));
461  break;
462  case 'Z':
463  currentCoord = static_cast<T*>(&(coords[2]));
464  break;
465  }
466 
467  coordId = gridChildId[c - 1];
468 
469  // quick transfer of data if same data types
470  if (sameType == true)
471  {
472  if (cgio_read_data(cgioNum, coordId, srcStart, srcEnd, srcStride, cellDim, memEnd, memStart,
473  memEnd, memStride, (void*)currentCoord))
474  {
475  char message[81];
476  cgio_error_message(message);
477  std::cerr << "cgio_read_data :" << message;
478  }
479  }
480  else
481  {
482  Y* dataArray = 0;
483  const cgsize_t memNoStride[3] = { 1, 1, 1 };
484 
485  // need to read into temp array to convert data
486  dataArray = new Y[nPts];
487  if (dataArray == 0)
488  {
489  std::cerr << "Error allocating buffer array\n";
490  break;
491  }
492  if (cgio_read_data(cgioNum, coordId, srcStart, srcEnd, srcStride, cellDim, memDims, memStart,
493  memDims, memNoStride, (void*)dataArray))
494  {
495  delete[] dataArray;
496  char message[81];
497  cgio_error_message(message);
498  std::cerr << "Buffer array cgio_read_data :" << message;
499  break;
500  }
501  for (vtkIdType ii = 0; ii < nPts; ++ii)
502  {
503  currentCoord[memStride[0] * ii] = static_cast<T>(dataArray[ii]);
504  }
505  delete[] dataArray;
506  }
507  }
508  return 0;
509 }
510 }
