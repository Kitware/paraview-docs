15 # Used to determine the version for ParaView source using "git describe", if git
16 # is found. On success sets following variables in caller's scope:
17 # ${var_prefix}_VERSION
18 # ${var_prefix}_VERSION_MAJOR
19 # ${var_prefix}_VERSION_MINOR
20 # ${var_prefix}_VERSION_PATCH
21 # ${var_prefix}_VERSION_PATCH_EXTRA
22 # ${var_prefix}_VERSION_FULL
23 # ${var_prefix}_VERSION_IS_RELEASE is true, if patch-extra is empty.
24 #
25 # If git is not found, or git describe cannot be run successfully, then these
26 # variables are left unchanged and status message is printed.
27 #
28 # Arguments are:
29 # source_dir : Source directory
30 # git_command : git executable
31 # var_prefix : prefix for variables e.g. "PARAVIEW".
32 function(determine_version source_dir git_command var_prefix)
33  if ("$Format:$" STREQUAL "")
34  # We are in an exported tarball and should use the shipped version
35  # information. Just return here to avoid the warning message at the end of
36  # this function.
37  return ()
38  elseif (NOT PARAVIEW_GIT_DESCRIBE AND
39  EXISTS ${git_command} AND
40  EXISTS ${source_dir}/.git)
41  execute_process(
42  COMMAND ${git_command} describe
43  WORKING_DIRECTORY ${source_dir}
44  RESULT_VARIABLE result
45  OUTPUT_VARIABLE output
46  ERROR_QUIET
47  OUTPUT_STRIP_TRAILING_WHITESPACE
48  ERROR_STRIP_TRAILING_WHITESPACE)
49  if (NOT result EQUAL 0)
50  # git describe failed (bad return code).
51  set(output "")
52  endif()
53  else ()
54  # note, output may be set to empty if PARAVIEW_GIT_DESCRIBE is not defined.
55  set(output "${PARAVIEW_GIT_DESCRIBE}")
56  endif()
57 
58  unset(tmp_VERSION)
59  extract_version_components("${output}" tmp)
60  if(DEFINED tmp_VERSION)
61  if (NOT "${tmp_VERSION}" STREQUAL "${${var_prefix}_VERSION}")
62  message(WARNING
63  "Version from git (${tmp_VERSION}) disagrees with hard coded version (${${var_prefix}_VERSION}). Either update the git tags or version.txt.")
64  endif()
65  foreach(suffix VERSION VERSION_MAJOR VERSION_MINOR VERSION_PATCH
66  VERSION_PATCH_EXTRA VERSION_FULL VERSION_IS_RELEASE)
67  set(${var_prefix}_${suffix} ${tmp_${suffix}} PARENT_SCOPE)
68  endforeach()
69  else()
70  message(STATUS
71  "Could not use git to determine source version, using version ${${var_prefix}_VERSION_FULL}")
72  endif()
73 endfunction()

75 # Extracts components from a version string. See determine_version() for usage.
76 function(extract_version_components version_string var_prefix)
77  string(REGEX MATCH "^v?(([0-9]+)\\.([0-9]+)\\.([0-9]+)-?(.*))$"
78  version_matches "${version_string}")
79  if(CMAKE_MATCH_0)
80  # note, we don't use CMAKE_MATCH_0 for `full` since it may or may not have
81  # the `v` prefix.
82  set(full ${CMAKE_MATCH_1})
83  set(major ${CMAKE_MATCH_2})
84  set(minor ${CMAKE_MATCH_3})
85  set(patch ${CMAKE_MATCH_4})
86  set(patch_extra ${CMAKE_MATCH_5})
87 
88  set(${var_prefix}_VERSION "${major}.${minor}" PARENT_SCOPE)
89  set(${var_prefix}_VERSION_MAJOR ${major} PARENT_SCOPE)
90  set(${var_prefix}_VERSION_MINOR ${minor} PARENT_SCOPE)
91  set(${var_prefix}_VERSION_PATCH ${patch} PARENT_SCOPE)
92  set(${var_prefix}_VERSION_PATCH_EXTRA ${patch_extra} PARENT_SCOPE)
93  set(${var_prefix}_VERSION_FULL ${full} PARENT_SCOPE)
94  if("${major}.${minor}.${patch}" VERSION_EQUAL "${full}")
95  set(${var_prefix}_VERSION_IS_RELEASE TRUE PARENT_SCOPE)
96  else()
97  set(${var_prefix}_VERSION_IS_RELEASE FALSE PARENT_SCOPE)
98  endif()
99  endif()
100 endfunction()
