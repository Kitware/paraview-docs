14 function (check_fortran_support)
15  if (DEFINED CMAKE_Fortran_COMPILER)
16  return ()
17  endif ()
18 
19  set(_desc "Looking for a Fortran compiler")
20  message(STATUS "${_desc}")
21  file(REMOVE_RECURSE "${CMAKE_CURRENT_BINARY_DIR}/CMakeFiles/CheckFortran")
22  file(WRITE "${CMAKE_CURRENT_BINARY_DIR}/CMakeFiles/CheckFortran/CMakeLists.txt"
23  "cmake_minimum_required(VERSION 3.3)
24 project(CheckFortran Fortran)
25 file(WRITE \"\${CMAKE_CURRENT_BINARY_DIR}/result.cmake\"
26 \"set(CMAKE_Fortran_COMPILER \\\"\${CMAKE_Fortran_COMPILER}\\\")\\n\"
27 \"set(CMAKE_Fortran_FLAGS \\\"\${CMAKE_Fortran_FLAGS}\\\")\\n\")\n")
28  execute_process(
29  WORKING_DIRECTORY "${CMAKE_CURRENT_BINARY_DIR}/CMakeFiles/CheckFortran"
30  COMMAND "${CMAKE_COMMAND}" . -G "${CMAKE_GENERATOR}"
31  OUTPUT_VARIABLE output
32  ERROR_VARIABLE output
33  RESULT_VARIABLE result)
34  include("${CMAKE_CURRENT_BINARY_DIR}/CMakeFiles/CheckFortran/result.cmake" OPTIONAL)
35  if (CMAKE_Fortran_COMPILER AND NOT result)
36  file(APPEND "${CMAKE_BINARY_DIR}${CMAKE_FILES_DIRECTORY}/CMakeOutput.log"
37  "${_desc} passed with the following output:\n"
38  "${output}\n")
39  else ()
40  set(CMAKE_Fortran_COMPILER NOTFOUND)
41  file(APPEND "${CMAKE_BINARY_DIR}${CMAKE_FILES_DIRECTORY}/CMakeError.log"
42  "${_desc} failed with the following output:\n"
43  "${output}\n")
44  endif ()
45  message(STATUS "${_desc} - ${CMAKE_Fortran_COMPILER}")
46  set(CMAKE_Fortran_COMPILER "${CMAKE_Fortran_COMPILER}" CACHE FILEPATH "Fortran compiler")
47  mark_as_advanced(CMAKE_Fortran_COMPILER)
48  set(CMAKE_Fortran_FLAGS "${CMAKE_Fortran_FLAGS}" CACHE STRING "Fortran flags")
49  mark_as_advanced(CMAKE_Fortran_FLAGS)
50 endfunction ()
